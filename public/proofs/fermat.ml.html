Fermat's Little Theorem (p = 3)

(*@ function rec power (x n: int) : int =
			if n > 0 then x * power x (n-1)
			else 1 *)	
(*@ requires n >= 0
    variant n *)

(*@ lemma power_three :
			forall x:int.
			power x 3 = x*x*x *)

let[@ghost] [@logic] rec little_fermat (x : int) : int = 
	if x = 0 then 0 else
    let z = little_fermat (x-1) in
    assert (power x 3 = power (x-1) 3 + 3 * ((x-1) * (x-1) + (x-1)) + 1);
    z + (x-1) * (x-1) + (x-1)
(*@ requires x >= 0
  	variant x
  	ensures power x 3 - x = 3 * result *)

let[@lemma] little_fermat_lemma (x : int) = 
		little_fermat x
(*@ requires x >= 0
		ensures exists y. power x 3 - x = 3 * y *)
