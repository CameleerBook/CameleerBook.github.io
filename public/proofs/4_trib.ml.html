Tribonacci Numbers

let[@ghost][@logic] rec trib n =
  if n = 0 then 0
  else if n <= 2 then 1
  else trib (n-1) + trib (n-2) + trib (n-3)
(*@ res = trib n
    requires n >= 0
    variant n *)

let tribonacci n =
  let z = ref 0 in
  let y = ref 1 in
  let x = ref 1 in
  for i = 0 to n - 1 do
    (*@ invariant !z = trib i
        invariant !y = trib (i+1)
        invariant !x = trib (i+2) *)
    let sum = !z + !y + !x in
    z := !y;
    y := !x;
    x := sum
  done;
  !z
(*@ res = tribonacci n
    requires n >= 0
    ensures res = trib n *)
