Binomial Coefficients (Dynamic Programming)

(*@ function rec bcoeff (n: int) (k: int) : int = 
    if k > n then 0 
    else if k = 0 || k = n then 1
    else bcoeff (n-1) (k-1) + bcoeff (n-1) k *)
(*@ requires 0 <= k /\ 0 <= n
    variant n *)

let binom n k =
  let a = Array.make (k+1) 0 in
  a.(0) <- 1;
  for i = 1 to n do
  (*@ invariant forall x. 0 <= x <= k -> a[x] = bcoeff (i-1) x *)
    let j = ref (min i k) in
    while !j > 0 do
    (*@ variant !j
        invariant !j <= min i k
        invariant forall x. 0 <= x <= !j -> a[x] = bcoeff (i-1) x
        invariant forall x. !j < x <= k -> a[x] = bcoeff i x *)
      a.(!j) <- a.(!j) + a.(!j-1);
      j := !j - 1;
    done;
  done;
  a.(k)
(*@ res = binom n k 
    requires 0 <= k <= n
    ensures res = bcoeff n k *)
