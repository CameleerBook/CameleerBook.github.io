Imperative Euclidean Division

(* Implementation adapted from MPRI Course 2-36-1: Proof of Program *)
(* Specification by Pedro Gasparinho, advised by MÃ¡rio Pereira *)

let euclidean_div x y =
	let r = ref x in
	let q = ref 0 in
	while !r >= y do
	(*@	invariant x = y * !q + !r
	  	invariant 0 <= !r
		  variant !r *)
		r := !r - y;
		q := !q + 1;
	done;
	(!q, !r)
(*@	(q, r) = euclidean_div x y
    requires x >= 0
    requires y > 0
    ensures x = y * q + r
    ensures 0 <= r < y *)
