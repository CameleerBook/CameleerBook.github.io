module type OrderedType = sig
	type t

	val eq: t -> t -> bool [@@logic]
	(*@ b = eq x y
		ensures b <-> x = y *)

	(*@ function le: t -> t -> bool *)

	(*@ axiom reflexive : forall x. le x x *)
	(*@ axiom total 		: forall x y. le x y \/ le y x *)
	(*@ axiom transitive: forall x y z. le x y -> le y z -> le x z *)

	val leq: t -> t -> bool [@@logic]
	(*@ b = leq x y
			ensures b <-> le x y *)
end

module MergeSort (Ord: OrderedType) = struct
	type elt = Ord.t

	let[@logic][@ghost] rec occ v = function
		| [] -> 0
		| h::t -> (if Ord.eq h v then 1 else 0) + occ v t
	(*@ res = occ v l
			variant l
			ensures 0 <= res <= List.length l
			ensures res > 0 <-> List.mem v l *)

	let[@lemma] rec occ_append (x: elt) (l1: elt list) (l2: elt list) =
		let p = occ x (l1 @ l2) = occ x l1 + occ x l2 in
		match l1 with
		| [] -> assert (p)
		| _::t -> occ_append x t l2; assert (p)
	(*@ occ_append x l1 l2
			variant l1
			ensures occ x (l1 @ l2) = occ x l1 + occ x l2 *) 

	(*@ predicate permut (l1 l2: elt list) =
		forall x. occ x l1 = occ x l2 *)

	(*@ predicate rec sorted (l: elt list) =
		match l with
		| [] | _::[] -> true
		| x::(y::ls) -> Ord.leq x y && sorted (y::ls) *)
	(*@ variant l *)

	let[@lemma] rec sorted_head (h: elt) (t: elt list) =
		match t with
		| [] -> ()
		| x::tt -> assert (Ord.leq h x); sorted_head h tt
	(*@ requires sorted (h::t)
			variant t
			ensures forall e. List.mem e t -> Ord.leq h e *)

	(*@ lemma sorted_cons :
		forall x: elt, l: elt list.
		(sorted l /\
		(forall e. List.mem e l -> Ord.leq x e)) <->
		sorted (x::l) *)

	let[@lemma] rec sorted_append (l1: elt list) (l2: elt list) =
		match l1 with
		| [] -> ()
		| h1::t1 -> 
			match l2 with
			| [] -> ()
			| h2::t2 -> assert (Ord.leq h1 h2); sorted_append l1 t2; sorted_append t1 l2
	(*@ sorted_append l1 l2
			requires sorted l1
			requires sorted l2
			requires forall x y. List.mem x l1 -> List.mem y l2 -> Ord.leq x y
			ensures sorted (l1 @ l2) *)

	let rec split (l: elt list) =
		match l with
		| [] -> ([], [])
		| x::ls -> 
			let ll, lr = split ls in (lr, x::ll)
	(*@ (r1, r2) = split l
		ensures permut l (r1 @ r2)
		ensures List.length r2 = List.length r1 \/ List.length r2 = List.length r1 + 1
		ensures List.length r2 + List.length r1 = List.length l
		variant List.length l *)

	let rec merge l1 l2 =
		match l1, l2 with
		| z, [] | [], z -> z
		| x::ls, y::rs -> 
		  if Ord.leq x y then x::(merge ls (y::rs))
		  else y::(merge (x::ls) rs)
	(*@ r = merge l1 l2
		requires sorted l1
		requires sorted l2
		ensures sorted r
		ensures permut r (l1 @ l2)
		variant List.length l1 + List.length l2 *)

	let rec merge_sort l =
		match l with
		| [] -> []
		| [x] -> [x]
		| _ -> let (a, b) = split l in
		merge (merge_sort a) (merge_sort b)
	(*@ r = merge_sort l
		ensures sorted r
		ensures permut r l
		variant List.length l *)
  
end
