Binary Search (Using Option Values)

(*@ predicate sorted (a: int array) = forall i j:int. 
    0 <= i <= j < Array.length a -> a.(i) <= a.(j) *)

let binary_search a v =
  let exception Break of int in 
  try
    let l = ref 0 in
    let h = ref (Array.length a - 1) in
    while !l <= !h do
    (*@ invariant 0 <= !l
        invariant !h < Array.length a
        invariant forall k. 0 <= k < !l \/ !h < k < Array.length a -> a.(k) <> v
        variant !h - !l *)
      let m = (!h + !l)/ 2 in
      if a.(m) < v then l := m+1
      else if a.(m) > v then h := m-1
      else raise (Break m)
    done;
    None
  with Break i -> Some i
(*@ r = binary_search a v
	  requires sorted a
    ensures match r with
      | None -> forall k. 0 <= k < Array.length a -> a.(k) <> v
      | Some i -> a.(i) = v *)
