McCarthy's 91 Function

let rec mccarthy n = 
	if n <= 100 then mccarthy(mccarthy (n + 11))
	else n - 10
(*@ r = mccarthy n
		variant 101 - n
		ensures n <= 100 -> r = 91
		ensures n > 100 -> r = n - 10 *)
