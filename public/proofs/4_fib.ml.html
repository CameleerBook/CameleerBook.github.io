(*@ function rec fib (n: int) : int =
    if n <= 1 then n
    else fib (n-1) + fib (n-2) *)
(*@ requires n >= 0
    variant n *)

let rec rec_fib n =
  if n <= 1 then n
  else rec_fib (n-1) + rec_fib (n-2)
(*@ res = rec_fib n
    requires n >= 0
    variant n
    ensures res = fib n *)    
    
let iter_fib n =
  let y = ref 0 in
  let x = ref 1 in
  for i = 0 to n - 1 do
  (*@ invariant !y = fib i
      invariant !x = fib (i+1) *)
    let sum = !y + !x in
    y := !x;
    x := sum
  done;
  !y
(*@ res = iter_fib n
    requires n >= 0
    ensures res = fib n *)
