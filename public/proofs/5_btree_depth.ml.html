type 'a tree = E | N of 'a tree * 'a * 'a tree

(*@ function rec preorder (t: 'a tree) : 'a list = 
    match t with
    | E -> []
    | N l x r -> x::(preorder l) @ (preorder r) *)
(*@ variant t *)

let rec dfs v = function
  | E -> false
  | N (l, x, r) -> x = v || dfs v l || dfs v r
(*@ r = dfs v t
    variant t
    ensures r <-> List.mem v (preorder t) *)
