Iterative Factorial

(*@ function rec fact (n: int) : int   =
    if n = 0 then 1
    else n * fact (n-1) *)
(*@ requires n >= 0 
    variant n *)

let rec factorial n =
  let r = ref 1 in
  for i = 1 to n do
  (*@ invariant !r = fact (i-1) *)  
    r := !r * i
  done;
  !r
(*@ r = factorial n
    requires n >= 0 
    ensures r = fact n *)
