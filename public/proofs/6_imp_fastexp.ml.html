Imperative Fast Exponentiation

(* Implementation adapted from MPRI Course 2-36-1: Proof of Program *)
(* Specification by Pedro Gasparinho, advised by MÃ¡rio Pereira *)

(*@ function rec power (x n: int) : int =
		if n = 0 then 1 
		else x * power x (n-1) *)
(*@ requires n >= 0
    variant n *)

let[@lemma] rec power_lemma (x: int) (n: int) = 
	if n > 1 then power_lemma x (n-2)
(*@ requires n >= 0
  	variant n
  	ensures mod n 2 = 0 -> power x n = (power (x * x) (div n 2)) 
	ensures mod n 2 = 1 -> power x n = x * (power (x * x) (div n 2)) *)

let fast_exp x n =
	let r = ref 1 in
	let p = ref x in
	let e = ref n in
	while !e > 0 do
	(*@ invariant !r * power !p !e = power x n
	    invariant !e >= 0
	    variant !e *)
		if !e mod 2 = 1 then r := !r * !p;
		p := !p * !p;
		e := !e / 2;
	done;
	!r
(*@ r = fast_exp x n
	requires n >= 0
    ensures r = power x n *)
