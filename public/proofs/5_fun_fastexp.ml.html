Functional Fast Exponentiation

   Proof of Program *)

(*@ function rec power (x n: int) : int =
		if n = 0 then 1 
		else x * power x (n-1) *)
(*@ requires n >= 0
    variant n *)

let[@lemma] rec power_lemma (x: int) (n: int) = 
	if n > 1 then power_lemma x (n-2)
(*@ requires n >= 0
  	variant n
  	ensures mod n 2 = 0 -> power x n = (power (x * x) (div n 2)) 
		ensures mod n 2 = 1 -> power x n = x * (power (x * x) (div n 2)) *)

let rec fast_exp x n =
	if n = 0 then 1
	else 
		let r = fast_exp (x * x) (n / 2) in
		if n mod 2 = 0 then r else x * r
(*@ r = fast_exp x n
		requires n >= 0
		variant n
		ensures r = power x n *)
