Functional Extended Euclidean Algorithm

let rec extended_gcd x y =
	if y = 0 then (1, 0, x)
	else
		let q = x / y in
		let (a, b, d) = extended_gcd y (x - q * y) in
		(b, a - q * b, d)
(*@ (a, b, d) = extended_gcd x y
		requires x <> 0
		variant abs y
		ensures d = a*x + b*y *)
